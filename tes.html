/** * ============================================================= * GAS Presensi Backend — Optimized version * Dynamic QR, Accelerometer, GPS Tracking * ============================================================= */ // Konfigurasi Global const CONFIG = { TOKEN_TTL_MS: 120000, SHEETS: { tokens: ["qr_token", "course_id", "session_id", "created_at", "expires_at", "used"], presence: ["presence_id", "user_id", "nim", "device_id", "course_id", "session_id", "qr_token", "ts", "recorded_at"], accel: ["device_id", "x", "y", "z", "sample_ts", "batch_ts", "recorded_at"], gps: ["device_id", "lat", "lng", "accuracy", "altitude", "ts", "recorded_at"] } }; /** * PINTU MASUK UTAMA (GET) */ function doGet(e) { const path = (e.parameter && e.parameter.path) || "dashboard"; const params = e.parameter; // Router Map untuk GET const routes = { "telemetry/gps/latest": () => apiHandler(() => handleGetGpsLatest(params.device_id)), "telemetry/gps/marker": () => apiHandler(() => handleGetGpsMarker(params.device_id)), "telemetry/gps/polyline": () => apiHandler(() => handleGetGpsPolyline(params)), "telemetry/accel/latest": () => apiHandler(() => handleGetAccelLatest(params.device_id)), "sensor/accel/latest": () => apiHandler(() => handleGetAccelLatest(params.device_id)), "telemetry/accel/history":() => apiHandler(() => handleGetAccelHistory(params)), "sensor/accel/history": () => apiHandler(() => handleGetAccelHistory(params)), "presence/status": () => apiHandler(() => handleGetPresenceStatus(params)), "presence/qr/status": () => apiHandler(() => handleCheckQRStatus(params.qr_token)), "admin/presence/list": () => apiHandler(() => handleGetPresenceList(params.course_id, params.session_id)), "sensor/gps/logs": () => apiHandler(() => handleGetAllGpsLogs()), // HTML Routes "dashboard": () => renderHtml("Dashboard", "Dashboard Utama"), "admin": () => renderHtml("Admin", "Admin Presensi QR"), "accelerometer": () => renderHtml("Accelerometer", "Sensor Accelerometer"), "client": () => renderHtml("Client", "Portal Presensi Mahasiswa") }; return routes[path] ? routes[path]() : sendError("Route not found", 404); } /** * PINTU MASUK UTAMA (POST) */ function doPost(e) { const path = (e.parameter && e.parameter.path) || ""; let body; try { body = JSON.parse(e.postData.contents); } catch (err) { return sendError("Invalid JSON body"); } // Router Map untuk POST const routes = { "presence/qr/generate": () => apiHandler(() => handleGenerateQR(body)), "presence/checkin": () => apiHandler(() => handleCheckin(body)), "telemetry/accel": () => apiHandler(() => handlePostAccelTelemetry(body)), "sensor/accel/batch": () => apiHandler(() => handlePostAccelTelemetry(body)), "telemetry/gps": () => apiHandler(() => handlePostGpsTelemetry(body)), "admin/presence/delete": () => apiHandler(() => handleDeletePresence(body)) }; return routes[path] ? routes[path]() : sendError("Route not found", 404); } // ============================================================= // HANDLERS: TELEMETRY (GPS & ACCELEROMETER) // ============================================================= function handlePostGpsTelemetry(body) { const { device_id, lat, lng, accuracy_m, accuracy, ts } = body; if (!device_id || lat === undefined || lng === undefined) throw new Error("Missing GPS data"); const sheet = getOrCreateSheet("gps"); sheet.appendRow([ String(device_id), lat, lng, accuracy_m || accuracy || null, null, ts || new Date().toISOString(), new Date().toISOString() ]); return { accepted: true, device_id }; } function handleGetGpsLatest(deviceId) { if (!deviceId) throw new Error("Missing device_id"); const data = getOrCreateSheet("gps").getDataRange().getValues(); for (let i = data.length - 1; i >= 1; i--) { if (String(data[i][0]) === deviceId) { return { ts: data[i][5], lat: data[i][1], lng: data[i][2], accuracy_m: data[i][3] }; } } throw new Error("No GPS data found"); } function handleGetGpsPolyline(params) { const { device_id, from, to } = params; if (!device_id || !from || !to) throw new Error("Missing device_id, from, or to"); const fromDate = new Date(from); const toDate = new Date(to); const data = getOrCreateSheet("gps").getDataRange().getValues(); const points = data.slice(1) .filter(row => { const ts = new Date(row[5]); return String(row[0]) === device_id && ts >= fromDate && ts <= toDate; }) .map(row => ({ lat: row[1], lng: row[2], ts: row[5] })); return { device_id, from, to, total_points: points.length, points }; } function handlePostAccelTelemetry(body) { const { device_id, ts, samples, data } = body; const accelData = samples || data; if (!device_id || !Array.isArray(accelData) || accelData.length === 0) throw new Error("Invalid samples"); const batchTs = ts || new Date().toISOString(); const recordedAt = new Date().toISOString(); const rows = accelData.map(s => [ String(device_id), Number(s.x || 0), Number(s.y || 0), Number(s.z || 0), String(s.t || s.ts || ""), batchTs, recordedAt ]); const sheet = getOrCreateSheet("accel"); sheet.getRange(sheet.getLastRow() + 1, 1, rows.length, rows[0].length).setValues(rows); return { accepted: rows.length, device_id }; } function handleGetAccelLatest(deviceId) { if (!deviceId) throw new Error("Missing device_id"); const data = getOrCreateSheet("accel").getDataRange().getValues(); for (let i = data.length - 1; i >= 1; i--) { if (String(data[i][0]) === deviceId) { return { t: data[i][4] || data[i][5], x: data[i][1], y: data[i][2], z: data[i][3] }; } } throw new Error("No accelerometer data found"); } function handleGetAccelHistory(params) { const deviceId = params.device_id; const limit = parseInt(params.limit) || 50; if (!deviceId) throw new Error("Missing device_id"); const data = getOrCreateSheet("accel").getDataRange().getValues(); let items = data.slice(1) .filter(row => String(row[0]) === deviceId) .map(row => ({ x: row[1], y: row[2], z: row[3], t: row[4] || row[5] })); if (items.length > limit) items = items.slice(-limit); return { device_id: deviceId, items }; } // ============================================================= // HANDLERS: PRESENCE & QR // ============================================================= function handleGenerateQR(body) { if (!body.course_id || !body.session_id) throw new Error("Missing course_id or session_id"); const now = new Date(); const token = "TKN-" + Utilities.getUuid().substring(0, 6).toUpperCase(); const expiresAt = new Date(now.getTime() + CONFIG.TOKEN_TTL_MS).toISOString(); getOrCreateSheet("tokens").appendRow([ token, body.course_id, body.session_id, now.toISOString(), expiresAt, "false" ]); return { qr_token: token, expires_at: expiresAt }; } function handleCheckin(body) { const { user_id, device_id, course_id, session_id, qr_token, ts } = body; if (!user_id || !qr_token || !course_id || !session_id) throw new Error("Missing required fields"); const tokenSheet = getOrCreateSheet("tokens"); const tokenData = tokenSheet.getDataRange().getValues(); // Cari token (dari bawah ke atas untuk performa) let tokenRec = null; for (let i = tokenData.length - 1; i >= 1; i--) { if (String(tokenData[i][0]) === qr_token) { tokenRec = { course_id: String(tokenData[i][1]), session_id: String(tokenData[i][2]), expires_at: tokenData[i][4] }; break; } } if (!tokenRec) throw new Error("token_invalid"); if (new Date(tokenRec.expires_at) < new Date()) throw new Error("token_expired"); if (course_id !== tokenRec.course_id || session_id !== tokenRec.session_id) throw new Error("token_course_session_mismatch"); const presSheet = getOrCreateSheet("presence"); const isDuplicate = presSheet.getDataRange().getValues().some(row => String(row[1]) === user_id && String(row[4]) === course_id && String(row[5]) === session_id ); if (isDuplicate) throw new Error("user_already_checked_in"); const presenceId = "PR-" + Utilities.getUuid().substring(0, 4).toUpperCase(); const recordedAt = new Date().toISOString(); presSheet.appendRow([ presenceId, user_id, user_id, device_id, course_id, session_id, qr_token, ts || recordedAt, recordedAt ]); return { presence_id: presenceId, status: "checked_in", user_id, course_id, session_id }; } function handleGetPresenceStatus(params) { const { user_id, course_id, session_id } = params; if (!user_id || !course_id || !session_id) throw new Error("Missing params"); const data = getOrCreateSheet("presence").getDataRange().getValues(); const record = data.find(r => String(r[1]).toLowerCase() === user_id.toLowerCase() && String(r[4]).toLowerCase() === course_id.toLowerCase() && String(r[5]).toLowerCase() === session_id.toLowerCase() ); return record ? { user_id, course_id, session_id, status: "checked_in", last_ts: record[7] } : { user_id, course_id, session_id, status: "not_checked_in", last_ts: null }; } function handleCheckQRStatus(token) { if (!token) throw new Error("Missing qr_token"); const data = getOrCreateSheet("tokens").getDataRange().getValues(); const found = data.find(r => String(r[0]) === token); if (!found) throw new Error("token_not_found"); return { qr_token: token, used: String(found[5]) === "true" }; } // ============================================================= // HANDLERS: ADMIN // ============================================================= function handleGetPresenceList(courseId, sessionId) { if (!courseId || !sessionId) throw new Error("Missing course_id or session_id"); const data = getOrCreateSheet("presence").getDataRange().getValues(); return data.slice(1) .filter(r => String(r[4]) === courseId && String(r[5]) === sessionId) .map(r => ({ presence_id: r[0], user_id: r[1], nim: r[2], device_id: r[3], course_id: r[4], session_id: r[5], qr_token: r[6], waktu: r[7], recorded_at: r[8] })); } function handleGetAllGpsLogs() { const data = getOrCreateSheet("gps").getDataRange().getValues(); return data.slice(1).reverse().map(r => ({ device_id: r[0], lat: r[1], lng: r[2], waktu: r[5] })); } function handleDeletePresence(body) { const { nim, course_id, session_id } = body; if (!nim || !course_id || !session_id) throw new Error("Missing nim, course_id, or session_id"); const sheet = getOrCreateSheet("presence"); const data = sheet.getDataRange().getValues(); for (let i = data.length - 1; i >= 1; i--) { if (String(data[i][2]) === nim && String(data[i][4]) === course_id && String(data[i][5]) === session_id) { sheet.deleteRow(i + 1); return { deleted: true, nim }; } } throw new Error("Record not found"); } function handleGetGpsMarker(deviceId) { if (!deviceId) throw new Error("Missing device_id"); const data = getOrCreateSheet("gps").getDataRange().getValues(); const row = data.reverse().find(r => String(r[0]) === deviceId); if (!row) throw new Error("No GPS data found"); return { device_id: row[0], lat: row[1], lng: row[2], accuracy: row[3], altitude: row[4], ts: row[5] }; } // ============================================================= // HELPERS (CORE UTILITIES) // ============================================================= /** * Membungkus eksekusi API untuk standardisasi response ok/error */ function apiHandler(fn) { try { const result = fn(); return sendSuccess(result); } catch (err) { return sendError(err.message); } } function getOrCreateSheet(name) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(name); if (!sheet) { sheet = ss.insertSheet(name); const headers = CONFIG.SHEETS[name]; if (headers) { sheet.getRange(1, 1, 1, headers.length).setValues([headers]); sheet.setFrozenRows(1); } } return sheet; } function sendSuccess(data) { return ContentService.createTextOutput(JSON.stringify({ ok: true, data: data })) .setMimeType(ContentService.MimeType.JSON); } function sendError(error, code = 400) { return ContentService.createTextOutput(JSON.stringify({ ok: false, error: error, code: code })) .setMimeType(ContentService.MimeType.JSON); } function renderHtml(file, title) { return HtmlService.createHtmlOutputFromFile(file) .setTitle(title) .addMetaTag("viewport", "width=device-width, initial-scale=1"); } function getAppUrl() { return ScriptApp.getService().getUrl();
